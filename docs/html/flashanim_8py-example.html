flashanim.py

An example of the usage of FlashSprite class. Flashing writings appears on the screen: you can pause them pressing SPACE and toggle persistent mode pressing LEFT ALT (i.e. they disappear or remain on the screen after flashing).

21import _setup
22from random import randrange
23import pygame
24import animimage
25
26# data for flashing writings
27TEXTS = ("Bonus", "Big win", "Next level", "Game over")
28FONTS = ("Arial", "Courier", "Times", "Verdana")
29SIZES = (36, 48, 60, 72)
30COLORS = ("yellow", "red", "dark green", "blue")
31
32pygame.init()
33screen = pygame.display.set_mode((800, 600))
34clk = pygame.time.Clock()
35CUSTOM_EV = pygame.event.custom_type()
36
37all_sprites = pygame.sprite.Group()
38
39# uncomment this to see debug info about sprites
40#animimage.set_debug(True)
41
42# set timer for triggering the 1st writing
43pygame.time.set_timer(CUSTOM_EV, randrange(500, 2000))
44
45# flags for paused and persistent mode
46paused, persistent = False, False
47# flag for exiting the main loop
48done = False
49# main loop
50while not done:
51 for ev in pygame.event.get():
52 if ev.type == pygame.QUIT:
53 done = True
54 elif ev.type == pygame.KEYDOWN:
55 # pressing SPACE toggles paused mode
56 if ev.key == pygame.K_SPACE:
57 if not paused:
58 # stop all animations, turning them on if invisible
59 # at this moment
60 for spr in all_sprites:
61 spr.anim_stop(visible=True)
62 else:
63 # restart all stopped animations
64 for spr in all_sprites:
65 spr.anim_start()
66 paused = not paused
67 # pressing ALT toggles persistent mode
68 elif ev.key == pygame.K_LALT:
69 if not persistent:
70 for spr in all_sprites:
71 # turn all animations into persistent FlashSprite
72 spr.hold=True
73 else:
74 # delete all sprites
75 all_sprites.empty()
76 persistent = not persistent
77 # triggers a new writing
78 elif ev.type == CUSTOM_EV:
79 if not paused:
80 # new FlashSprite
81 anim = animimage.FlashSprite()
82 # choose a random font, size, text and color and render it into a pygame Surface
83 fnt = pygame.font.SysFont(FONTS[randrange(4)], SIZES[randrange(4)], bold=True, italic=randrange(2))
84 surf = fnt.render(TEXTS[randrange(4)], True, COLORS[randrange(4)])
85 # set the Surface as the FlashSprite image: this starts the animation
86 anim.set_image(surf)
87 # choose a random position for the writing
88 anim.rect.topleft = (randrange(20, 800 - surf.get_width() - 20),
89 randrange(20, 600 - surf.get_height() - 20))
90 # check if the writing overlaps another animation (only if not in persistent mode)
91 if not pygame.sprite.spritecollide(anim, all_sprites, False) or persistent:
92 # add the animation to the main Group, so it can be drawn
93 all_sprites.add(anim)
94 # set a random speed for flashing and persistent on if in persistent mode
95 anim.set_param(rate=randrange(10, 18) / 2, flashes=randrange(4, 9), hold=persistent)
96 pygame.time.set_timer(CUSTOM_EV, randrange(2000, 3000))
97
98 screen.fill("aqua")
99 # make all animation progress
100 all_sprites.update()
101 all_sprites.draw(screen)
102 pygame.display.flip()
103 clk.tick(30)
104
105pygame.quit()
