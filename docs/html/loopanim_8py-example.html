loopanim.py

An example of the usage of AnimSprite class with looped animations. Cars move from left to right: you can pause animations (not moving) pressing SPACE and change their speed with LEFT and RIGHT arrow keys.

26pygame.init()
27screen = pygame.display.set_mode((800, 600))

31all_images = slicesheet.slicesheet(join("..", "carro.png"), 10, 4, 2380)
32images = (all_images[0:10], all_images[10:20], all_images[20:30], all_images[30:40])
33all_sprites = pygame.sprite.Group()
34
35# uncomment this to see debug info about sprites
36#animimage.set_debug(True)
37
38# the speed of the cars in pixel per 30th of sec. It can vary from 1.0 to 4.0 at
39# steps of 0.5
40speed = 2.0
41# the speed of animations in 30th of sec per frame (see animimage.set_rate())
42anim_rate = 4 / speed
43# internal variable (see below)
44offset = 0.0
45print("speed = {:2.1f} anim_rate = {:4.2f}".format(speed, anim_rate))
46
47# set timer for triggering the 1st car
48pygame.time.set_timer(CUSTOM_EV, randrange(500, 2000))
49
50# flag for paused mode
51paused = False
52# flag for exiting the main loop
53done = False
54# main loop
55while not done:
56 for ev in pygame.event.get():
57 if ev.type == pygame.QUIT:
58 done = True
59 elif ev.type == pygame.KEYDOWN:
60 # pressing SPACE toggles paused mode
61 if ev.key == pygame.K_SPACE:
62 if not paused:
63 # stop all animations
64 for spr in all_sprites:
65 spr.anim_stop()
66 else:
67 # restart all stopped animations
68 for spr in all_sprites:
69 spr.anim_start()
70 paused = not paused
71 # pressing left and right arrows increments or decrements car speed
72 elif ev.key in (pygame.K_LEFT, pygame.K_RIGHT):
73 # set new car speed
74 speed = max(1.0, speed - 0.5) if ev.key == pygame.K_LEFT else min(4.0, speed + 0.5)
75 # update animation speed
76 anim_rate = 4 / speed
77 # reset offset
78 offset = 0.0
79 for spr in all_sprites:
80 spr.set_rate(anim_rate)
81 print("speed = {:2.1f} anim_rate = {:4.2f}".format(speed, anim_rate))
82 # triggers a new car
83 elif ev.type == CUSTOM_EV:
84 # new AnimSprite
85 anim = animimage.AnimSprite()
86 # choose a random car (four different colours)
87 anim.set_images(images[randrange(4)], True)
88 # choose a random height for the car
89 anim.rect.topleft = (-anim.rect.width, randrange(150, 400))
90 # check if the car overlaps another animation
91 if not pygame.sprite.spritecollide(anim, all_sprites, False):
92 # if not add the animation to the main Group, so it can be drawn
93 all_sprites.add(anim)
94 # set the animation rate
95 anim.set_rate(anim_rate)
96 pygame.time.set_timer(CUSTOM_EV, randrange(2000, 6000))
97
98 # if you have a non-integer speed you must alternate between speed - 0.5 and
99 # speed + 0.5. So we need the variable offset tp get this
100 offset += (speed % 1)
101 for sp in all_sprites:
102 sp.rect.x += (speed + (0.5 if offset == 1 else -0.5 if offset == 0.5 else 0.0))
103 if sp.rect.x > screen.get_width():
104 sp.kill()
105 if offset == 1.0:
106 offset = 0.0
107
108 screen.fill("aqua")
109 # make all animation progress
110 all_sprites.update()
111 all_sprites.draw(screen)
112 pygame.display.flip()
113 clk.tick(30)
